\begin{frame}{Proposal and contributions}
	\begin{itemize}[<+->]
		\item \textbf{Semantic OS system} for indoor environment
		\item It searches for a specific door sign (goal-door)
		\item It was evaluated in simulated and real indoor environments, compared against other OS systems and humans
		\item The \textbf{novelties} of this work:
		\begin{itemize}[<+->]
			\item Target object and scenario
			\item Textual information and organization of the environment
			\item Analysis that highlights the advantages of the use of semantic information
		\end{itemize}
	\end{itemize}
\end{frame}
%========================================== ############# ======================================================
%========================================== ############# ======================================================
%========================================== ############# ======================================================
\begin{frame}{System overview}
	\begin{itemize}[<+->]
		\item Our system is composed by:
		\begin{itemize}[<+->]
			\item Mapping 
			\item Image Processing
			\item Map Segmentation
			\item Semantic Planner
		\end{itemize}
	\end{itemize}
\end{frame}
%========================================== ############# ======================================================        
%========================================== ############# ======================================================
%========================================== ############# ======================================================
\begin{frame}{Mapping module}
	\begin{itemize}[<+->]
		\item A 2D grid map is build ({\tiny Simulation and real worlds})
		\item Voronoi diagram 
		\item Visited regions
		\item Boundary Value Problem (BVP)
	\end{itemize}
\end{frame}
%========================================== ############# ======================================================
%========================================== ############# ======================================================
%========================================== ############# ======================================================
\begin{frame}{Map Segmentation module}
	\begin{itemize}[<+->]
		\item Simultaneously to the Mapping module
		\item Divides the map based on the free space
		\item Kernel Density Estimation (KDE)
		\item An abstraction of the 2D grid map for grouping the detected door signs
	\end{itemize}
\end{frame}
%========================================== ############# ======================================================
%========================================== ############# ======================================================
%========================================== ############# ======================================================
\begin{frame}{Image Processing module}
	\begin{itemize}[<+->]
		\item Optical character recognition (OCR)
		\item Recognize the number of door signs in RGB images
		\item We use the work proposed by XXXXX
		\item The recognized numbers are associated to the map segments
	\end{itemize}
\end{frame}
%========================================== ############# ======================================================
%========================================== ############# ======================================================
%========================================== ############# ======================================================
\begin{frame}{Semantic Planner module}
	\begin{itemize}[<+->]
		\item The core of our semantic OS system
		\item It estimates which regions are more promising to find the goal-door
		\item It is composed of five different factors:
		\begin{itemize}[<+->]
			\item Growing Direction
			\item Parity
			\item Doors Orientation
			\item Robot Orientation
			\item Distance			
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Summary}
	\begin{columns}[T]
		\column{0.55\textwidth}
		\begin{itemize}[<+->]
			\item Semantic OS system that:
			\begin{itemize}
				\item  relies on semantic information inferred from text within the environment
				\item  does not require that the door signs are set according to a specific pattern
				\item  performs better than the greedy OS system and humans participants
			\end{itemize}
			\item It is possible to expand the robot's perception with semantic information
			\item Analysis of the difficulty of the OS problem by the experiments with human participants
		\end{itemize}
		\column{0.55\textwidth}
	\end{columns}
