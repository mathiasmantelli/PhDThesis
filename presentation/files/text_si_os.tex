\begin{frame}{Proposal and contributions}
	\begin{itemize}[<+->]
		\item \textbf{Semantic OS system} for indoor environment
		\item It searches for a specific door sign (goal-door)
		\item It was evaluated in simulated and real indoor environments, compared against other OS systems and humans
		\item The \textbf{novelties} of this work:
		\begin{itemize}[<+->]
			\item \textbf{Target object} and scenario
			\item \textbf{Textual information} and \textbf{organization of the environment}
			\item Analysis that highlights the \textbf{advantages} of the use of \textbf{semantic information}
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Summary}
	\begin{columns}[T]
		\column{0.55\textwidth}
		\begin{itemize}[<+->]
			\item Semantic OS system that:
			\begin{itemize}
				\item  relies on semantic information inferred from text within the environment
				\item  does not require that the door signs are set according to a specific pattern
				\item  performs better than the greedy OS system and humans participants
			\end{itemize}
			\item It is possible to expand the robot's perception with semantic information
			\item Analysis of the difficulty of the OS problem by the experiments with human participants
		\end{itemize}
		\column{0.55\textwidth}
	\end{columns}
