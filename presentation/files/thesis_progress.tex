\begin{frame}{Conclusion}
	\centering
	\begin{itemize}[<+->]
		\item \textbf{Investigated} the\textbf{ usage of semantic information} associated with the \textbf{spatial} and \textbf{temporal organisation} of the \textbf{environment} in the context of \textbf{OS problems}
		\item \textbf{Expanded} and \textbf{improved} the \textbf{SR's perception} to perform \textbf{OS tasks} in everyday living spaces
		\item Overcomed the \textbf{limitations} of the purely \textbf{geometric perception} for \textbf{OS tasks} inspired by human reasoning
		\item Shown the \textbf{two sources} of data in which the \textbf{semantic information} was \textbf{inferred} from:
		\begin{itemize}
			\item \textbf{Numbers - NSOS system}
			\item \textbf{Semi-dynamic objects - LSOS system}
		\end{itemize}
	\end{itemize}
\end{frame}
