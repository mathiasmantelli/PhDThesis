\begin{frame}{Discussion}
	\centering
	\begin{itemize}[<+->]
		\item \textbf{Investigated} the\textbf{ usage of semantic information} associated with the \textbf{spatial} and \textbf{temporal organization} of the \textbf{environment} in the context of \textbf{OS problems}
		\item \textbf{Expand} and \textbf{improve} the \textbf{robot's perception} to perform \textbf{OS tasks} in everyday living spaces
		\item The \textbf{limitations} of the purely \textbf{geometric perception} for \textbf{OS tasks}
		\item \textbf{Two sources} of data in which the \textbf{semantic information} was \textbf{inferred} from:
		\begin{itemize}
			\item \textbf{Numbers - NSOS system}
			\item \textbf{Semi-dynamic objects - LSOS system}
		\end{itemize}
